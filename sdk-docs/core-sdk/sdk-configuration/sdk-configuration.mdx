### Example: Default configuration

```cs
import { configureEnvironment } from "@funkit/core"

const apiKey = "ZrhepzWGxm74D0sqstuhT6dGrJxhoy8SZIToX6I3"
await configureEnvironment({
  chain: "goerli",
  apiKey,
})
```

### Example: Gasless configuration

```cs
import { configureEnvironment } from "@funkit/core"

const apiKey = "ZrhepzWGxm74D0sqstuhT6dGrJxhoy8SZIToX6I3"
await configureEnvironment({
  chain: "goerli",
  apiKey,
  gasSponsor: {
    sponsorAddress: '0xCB5D0b4569A39C217c243a436AC3feEe5dFeb9Ad',
  },
})
```

### Example: ERC-20 gas payments configuration

import { configureEnvironment } from "@funkit/core"

const apiKey = "ZrhepzWGxm74D0sqstuhT6dGrJxhoy8SZIToX6I3"

  chain: "goerli",
  apiKey,
  gasSponsor: {
    sponsorAddress: '0x40C0cCa76088D45106c2D74D0B4B6405865f22De',
